#version 330 core
out vec4 FragColor;
in vec3 vertexColor;
in vec2 TexCoord;

uniform sampler2D texture0;
uniform sampler2D texture1;
uniform float lever = 0.5;

void main()
{
	// FragColor = vertexColor;
	// FragColor = globalColor;
	// FragColor = vec4(1.0f);
	// FragColor = texture(texture0, TexCoord) * vec4(vertexColor, 1.0);
	FragColor = mix(texture(texture0, TexCoord), texture(texture1, TexCoord), lever);
}
